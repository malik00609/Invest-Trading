import random import time # پچھلے راؤنڈز کا ڈیٹا history = [] # فرضی پریڈکشن فنکشن def predict_next(history): if len(history) < 3: return round(random.uniform(1.0, 2.0), 2) avg = sum(history[-3:]) / 3 prediction = avg + random.uniform(-0.5, 0.5) return round(max(1.0, prediction), 2) # ہر راؤنڈ میں نیا ریزلٹ بنائیں def generate_result(): chance = random.random() if chance < 0.1: return round(random.uniform(1.0, 1.5), 2) elif chance < 0.9: return round(random.uniform(1.5, 3.0), 2) else: return round(random.uniform(3.0, 100.0), 2) # مین فنکشن def main(): print("🚀 Aviator-style Prediction Demo Tool (صرف لرننگ کے لیے)") rounds = 20 for i in range(rounds): print(f"\n🎮 راؤنڈ {i+1}") predicted = predict_next(history) print(f"🔮 پیشن گوئی: {predicted}x") input("▶ جاری رکھنے کے لیے Enter دبائیں...") result = generate_result() history.append(result) print(f"📊 اصل نتیجہ: {result}x") time.sleep(1) if __name__ == "__main__": main()
